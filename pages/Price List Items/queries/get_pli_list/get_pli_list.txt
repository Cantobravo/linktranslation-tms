SELECT
  pli.price_list_item_id,
  pli.price_list_id,
  pl.price_list_name,
  pl.price_list_type,
  pl.currency_id,                    -- from price_lists (pli has no currency_id)
  pli.service_id,
  s.service_name,
  pli.unit_id,
  u.unit_name,
  pli.source_language_id,
  sl.language_name  AS source_language_name,
  sl.language_acronym AS source_language_acr,
  pli.target_language_id,
  tl.language_name  AS target_language_name,
  tl.language_acronym AS target_language_acr,
  pli.rate,
  pli.effective_start_date,
  pli.effective_end_date,
  pli.item_is_active,
  pli.note,
  pli.created_at,
  pli.updated_at
FROM price_list_items AS pli
JOIN price_lists   AS pl ON pl.price_list_id = pli.price_list_id
JOIN services      AS s  ON s.service_id = pli.service_id
JOIN units         AS u  ON u.unit_id = pli.unit_id
JOIN languages     AS sl ON sl.language_id = pli.source_language_id
JOIN languages     AS tl ON tl.language_id = pli.target_language_id
ORDER BY pli.updated_at DESC;
