SELECT
  cs.customer_service_id,
  cs.customer_id,
  c.customer_name,
  cs.customer_account_id,
  ca.customer_account_name,
  cs.service_id,
  s.service_name,
  cs.currency_id,
  cur.currency_name,
  cs.unit_id,
  u.unit_name,
  cs.source_language_id,
  sl.language_name AS source_language_name,
  cs.target_language_id,
  tl.language_name AS target_language_name,
  cs.amount,
  cs.note,
  cs.created_at,
  cs.updated_at
FROM customer_services cs
JOIN customers c         ON c.customer_id          = cs.customer_id
LEFT JOIN customer_accounts ca ON ca.customer_account_id = cs.customer_account_id
LEFT JOIN services s          ON s.service_id           = cs.service_id
LEFT JOIN currencies cur      ON cur.currency_id        = cs.currency_id
LEFT JOIN units u             ON u.unit_id              = cs.unit_id
LEFT JOIN languages sl        ON sl.language_id         = cs.source_language_id
LEFT JOIN languages tl        ON tl.language_id         = cs.target_language_id
ORDER BY cs.customer_service_id DESC;
