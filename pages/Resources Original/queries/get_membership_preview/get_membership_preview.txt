SELECT
  r.resource_id,
  r.resource_full_name,
  CASE WHEN COALESCE(wtm.work_team_member_id, 0) = 0 THEN 0 ELSE 1 END AS already_in_team
FROM resources r
LEFT JOIN work_team_members wtm
  ON wtm.resource_id = r.resource_id
 AND wtm.work_team_id = {{ i_wt_pick.selectedOptionValue || 0 }}
WHERE r.resource_id IN (
  {{ (tbl_resources.selectedRows || []).map(x => x.resource_id).join(',') || '0' }}
);
