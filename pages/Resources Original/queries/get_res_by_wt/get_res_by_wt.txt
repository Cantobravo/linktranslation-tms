SELECT
  r.resource_id,
  r.resource_full_name,
  r.resource_email,
  r.resource_status,
  wtm.work_team_id,
  wt.work_team_name
FROM work_team_members wtm
JOIN resources r ON r.resource_id = wtm.resource_id
JOIN work_teams wt ON wt.work_team_id = wtm.work_team_id
WHERE COALESCE(wtm.is_active, 1) = 1
  AND wtm.work_team_id = COALESCE({{ sel_wt_filter.selectedOptionValue || null }}, wtm.work_team_id)
ORDER BY r.resource_full_name;
