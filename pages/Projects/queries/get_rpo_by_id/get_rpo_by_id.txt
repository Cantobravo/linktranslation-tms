SELECT
  rpo.purchase_order_id,
  rpo.project_id,
  rpo.resource_id,
  rpo.currency_id,
  rpo.service_id,
  rpo.source_language_id,
  rpo.target_language_id,
  rpo.unit_id,
  rpo.volume,
  rpo.rate,
  rpo.total,
  rpo.date,
  rpo.due_date,
  rpo.note,
  r.resource_email
FROM resource_purchase_orders rpo
LEFT JOIN resources r ON r.resource_id = rpo.resource_id
WHERE rpo.purchase_order_id = {{ appsmith.store.rpo_id || null }};
