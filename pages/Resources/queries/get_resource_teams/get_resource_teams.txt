SELECT
  wtm.work_team_member_id,
  wtm.resource_id,
  wtm.work_team_id,
  wt.work_team_name,
  wtm.is_available,
  COALESCE(wtm.is_active, 1) AS is_active,
  wtm.created_at,
  wtm.updated_at
FROM work_team_members wtm
JOIN work_teams wt
  ON wt.work_team_id = wtm.work_team_id
 AND COALESCE(wt.is_active, 1) = 1
WHERE wtm.resource_id = {{
  tbl_resources.selectedRow?.resource_id
  || (tbl_resources.selectedRows?.[0]?.resource_id)
  || appsmith.store.resRow?.resource_id
  || 0
}}
ORDER BY wt.work_team_name;
