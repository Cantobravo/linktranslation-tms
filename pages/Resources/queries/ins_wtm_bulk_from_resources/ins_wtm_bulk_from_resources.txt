INSERT INTO work_team_members
(work_team_id, resource_id, is_available, certification_status, daily_capacity_words, po_status, rates_text, note)
SELECT
  {{ i_wt_pick.selectedOptionValue || 0 }}  AS work_team_id,
  r.resource_id,
  'Unknown'  AS is_available,
  'None'     AS certification_status,
  NULL       AS daily_capacity_words,
  NULL       AS po_status,
  NULL       AS rates_text,
  NULL       AS note
FROM resources r
LEFT JOIN work_team_members w
  ON w.work_team_id = {{ i_wt_pick.selectedOptionValue || 0 }}
 AND w.resource_id  = r.resource_id
WHERE r.resource_id IN (
  {{ (tbl_resources.selectedRows || []).map(x => x.resource_id).join(',') || '0' }}
)
AND COALESCE(w.work_team_member_id, 0) = 0;
