SELECT
  wtmc.wtm_cred_id,
  wtmc.username,
  wtmc.alias_email,
  wtmc.password_text
FROM work_team_member_credentials wtmc
JOIN work_team_members wtm
  ON wtm.work_team_member_id = wtmc.work_team_member_id
WHERE
  COALESCE(wtmc.is_active, 1) = 1
  AND wtm.resource_id = {{ appsmith.store.resRow?.resource_id || tbl_resources.selectedRow?.resource_id || 0 }}
ORDER BY wtmc.username
