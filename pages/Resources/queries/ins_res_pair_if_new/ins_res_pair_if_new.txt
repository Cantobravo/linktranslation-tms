INSERT INTO resource_language_pairs
  (resource_id, source_language_id, target_language_id, is_active, created_at)
SELECT
  {{ this.params.resource_id }},
  {{ this.params.source_language_id }},
  {{ this.params.target_language_id }},
  1,
  NOW()
FROM DUAL
WHERE NOT EXISTS (
  SELECT 1
  FROM resource_language_pairs
  WHERE resource_id = {{ this.params.resource_id }}
    AND source_language_id = {{ this.params.source_language_id }}
    AND target_language_id = {{ this.params.target_language_id }}
);
