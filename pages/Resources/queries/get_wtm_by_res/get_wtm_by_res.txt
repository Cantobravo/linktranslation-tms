SELECT
  MIN(wtm.work_team_member_id) AS work_team_member_id,
  r.resource_full_name
FROM work_team_members wtm
LEFT JOIN resources r ON r.resource_id = wtm.resource_id
WHERE
  wtm.resource_id = {{ appsmith.store.resRow?.resource_id || tbl_resources.selectedRow?.resource_id || 0 }}
  AND COALESCE(wtm.is_active, 1) = 1
GROUP BY
  r.resource_full_name
ORDER BY
  r.resource_full_name
