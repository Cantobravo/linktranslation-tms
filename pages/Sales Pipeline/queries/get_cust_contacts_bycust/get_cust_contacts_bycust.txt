SELECT
  cc.customer_contact_id AS contact_id,
  cc.customer_id,
  cc.customer_contact_name AS contact_full_name,
  cc.contact_position AS contact_role,
  cc.customer_contact_email AS contact_email,
  cc.customer_contact_first_contact AS initial_contact_date,
  cc.customer_contact_last_contact AS last_contact_date,
  cc.customer_contact_next_contact AS next_contact_date,
  cc.customer_contact_status AS contact_status,
  cc.customer_contact_notes
FROM customer_contacts cc
WHERE cc.customer_id = {{
  Number(
    appsmith.store.sel_cust_id
    || (tbl_pipeline.selectedRow ? tbl_pipeline.selectedRow.customer_id : null)
  ) || -1
}}
ORDER BY cc.customer_contact_name;
