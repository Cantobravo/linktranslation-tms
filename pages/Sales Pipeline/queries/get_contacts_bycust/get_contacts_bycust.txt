SELECT
  cc.customer_contact_id,
  cc.customer_contact_name,
  cc.contact_position,
  cc.customer_contact_email,
  cc.customer_id,
  c.customer_name,
  cc.customer_contact_industry,
  cc.customer_contact_website,
  cc.customer_contact_linkedin,
  cc.customer_contact_first_contact,
  cc.customer_contact_last_contact,
  cc.customer_contact_next_contact,
  cc.customer_contact_contact_method,
  cc.customer_contact_test,
  cc.customer_contact_notes,
  cc.contact_is_active,
  cc.created_at,
  cc.updated_at
FROM customer_contacts cc
LEFT JOIN customers c
  ON c.customer_id = cc.customer_id
WHERE
  cc.customer_id = COALESCE({{ appsmith.store.sel_cust_id }}, -1)
  AND cc.contact_is_active = 1
ORDER BY cc.customer_contact_name
