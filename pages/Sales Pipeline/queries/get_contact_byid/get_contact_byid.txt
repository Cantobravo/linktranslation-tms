SELECT
  customer_contact_id       AS contact_id,
  customer_id,
  customer_contact_name     AS contact_full_name,
  contact_position          AS contact_role,
  customer_contact_email    AS contact_email,
  customer_contact_status   AS contact_status,
  customer_contact_industry,
  customer_contact_website,
  customer_contact_linkedin,
  customer_contact_first_contact AS initial_contact_date,
  customer_contact_last_contact  AS last_contact_date,
  customer_contact_next_contact  AS next_contact_date,
  customer_contact_contact_method,
  customer_contact_test,
  customer_contact_notes
FROM customer_contacts
WHERE customer_contact_id = {{ appsmith.store.upd_contact_id || -1 }};
