SELECT
  p.customer_portal_id AS portal_id,
  p.customer_portal_name AS portal_name,
  p.customer_portal_first_update AS first_update,
  p.customer_portal_last_update AS last_update,
  p.customer_portal_next_update AS next_update
FROM customer_portals p
WHERE p.customer_portal_id = {{
  Number(
    appsmith.store.sel_portal_id
    || (tbl_cust_portals.selectedRow ? tbl_cust_portals.selectedRow.portal_id : null)
  ) || -1
}};
