SELECT
  cc.customer_contact_id AS contact_id,
  cc.customer_contact_name AS contact_full_name,
  cc.customer_contact_first_contact AS initial_contact,
  cc.customer_contact_last_contact AS last_contact,
  cc.customer_contact_next_contact AS next_contact
FROM customer_contacts cc
WHERE cc.customer_contact_id = {{
  Number(
    appsmith.store.sel_contact_id
    || (tbl_cust_contacts.selectedRow ? tbl_cust_contacts.selectedRow.contact_id : null)
  ) || -1
}};
