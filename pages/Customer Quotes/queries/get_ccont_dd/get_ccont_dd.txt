SELECT
  cc.customer_contact_id AS value,
  CONCAT(c.customer_name, ' â€” ', cc.customer_contact_name) AS label,
  cc.customer_id
FROM customer_contacts cc
JOIN customers c ON c.customer_id = cc.customer_id
WHERE COALESCE(cc.customer_contact_status,'Active') <> 'Inactive'
  AND COALESCE(c.customer_status,'Active') <> 'Inactive'
ORDER BY c.customer_name, cc.customer_contact_name;
