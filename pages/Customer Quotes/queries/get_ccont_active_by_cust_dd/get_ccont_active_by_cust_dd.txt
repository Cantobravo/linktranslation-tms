SELECT
  cc.customer_contact_id AS value,
  cc.customer_contact_name AS label,
  cc.customer_id
FROM customer_contacts cc
WHERE COALESCE(cc.customer_contact_status, 'Active') = 'Active'
  AND cc.customer_id = COALESCE({{ j_cquote_ins_jsn.formData.customer_id ?? null }}, cc.customer_id)
ORDER BY cc.customer_contact_name;
