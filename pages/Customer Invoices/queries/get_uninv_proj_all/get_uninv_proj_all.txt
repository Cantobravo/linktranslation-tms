SELECT
  p.project_id,
  p.project_name,
  CAST(p.project_rate  AS DECIMAL(10,2)) AS project_rate,
  CAST(p.project_total AS DECIMAL(10,2)) AS project_total,
  p.project_start_date,
  p.project_end_date,
  COALESCE(p.project_status, 'Not Invoiced') AS project_status,
  c.customer_id,
  c.customer_name
FROM projects p
JOIN customers c ON c.customer_id = p.customer_id
WHERE COALESCE(p.project_status, 'Not Invoiced') = 'Not Invoiced'
ORDER BY c.customer_name, p.project_end_date DESC, p.project_id DESC;
