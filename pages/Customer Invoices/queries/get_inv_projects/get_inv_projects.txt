SELECT
  p.project_id,
  p.project_name,
  p.project_status,
  CAST(p.project_rate  AS DECIMAL(10,2)) AS project_rate,
  CAST(p.project_total AS DECIMAL(10,2)) AS project_total,
  p.project_start_date,
  p.project_end_date
FROM invoice_project_link l
JOIN projects p ON p.project_id = l.project_id
WHERE l.customer_invoice_id = {{ appsmith.store.invId }}
ORDER BY p.project_end_date DESC, p.project_id DESC;
