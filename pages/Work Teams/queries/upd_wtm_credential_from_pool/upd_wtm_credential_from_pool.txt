UPDATE work_team_member_credentials w
JOIN work_team_un_credentials u
  ON u.wtu_cred_id = {{ i_uncred_pick_upd.selectedOptionValue }}
SET
  w.system_code      = u.system_code,
  w.username         = u.username,
  w.alias_email      = u.alias_email,
  w.password_text    = u.password_text,
  w.cat_profile      = u.cat_profile,
  w.origin_uncred_id = u.wtu_cred_id,
  w.updated_at       = NOW()
WHERE w.wtm_cred_id = {{ tbl_wtm_creds.triggeredRow.wtm_cred_id }};
