SELECT
  wtmc.wtm_cred_id,
  wtmc.system_code,
  wtmc.username,
  wtmc.alias_email,
  wtmc.password_text,
  wtmc.cat_profile,
  wtmc.origin_uncred_id,            -- ‚Üê add this
  COALESCE(wtmc.is_active,1) AS is_active,
  DATE_FORMAT(wtmc.created_at, '%Y-%m-%d %H:%i') AS created_at
FROM work_team_member_credentials wtmc
WHERE wtmc.work_team_member_id = {{ appsmith.store.wtmRow?.work_team_member_id || 0 }}
  AND COALESCE(wtmc.is_active,1) = 1
ORDER BY wtmc.wtm_cred_id DESC;
