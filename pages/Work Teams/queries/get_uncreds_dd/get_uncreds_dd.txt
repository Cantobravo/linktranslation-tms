SELECT
  wtu_cred_id                      AS value,
  CONCAT(COALESCE(system_code,''), ' · ', username,
         CASE WHEN COALESCE(alias_email,'') <> '' THEN CONCAT(' · ', alias_email) ELSE '' END
  )                                AS label,
  system_code,
  username,
  alias_email,
  password_text,
  cat_profile
FROM work_team_un_credentials
WHERE is_assigned = 0
ORDER BY system_code, username;
