SET

  link_url   = COALESCE(NULLIF({{ j_wtm_link_upd_form.formData.link_url }}, ''), link_url),
  link_label = NULLIF({{ j_wtm_link_upd_form.formData.link_label }}, ''),
  wordcount  = NULLIF({{ j_wtm_link_upd_form.formData.wordcount }}, ''),
  hours      = NULLIF({{ j_wtm_link_upd_form.formData.hours }}, ''),
  note       = NULLIF({{ j_wtm_link_upd_form.formData.note }}, ''),
  updated_at = NOW()
WHERE wtm_link_id = {{ (appsmith.store.wtmLinkRow?.wtm_link_id) || tbl_wtm_links.triggeredRow.wtm_link_id }};
