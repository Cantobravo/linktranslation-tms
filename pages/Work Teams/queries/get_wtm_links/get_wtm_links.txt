SELECT
  wtmf.wtm_link_id,
  wtmf.link_url,
  wtmf.link_label,
  wtmf.wordcount,
  wtmf.hours,
  wtmf.note,
  COALESCE(wtmf.is_active,1) AS is_active,
  DATE_FORMAT(wtmf.created_at, '%Y-%m-%d %H:%i') AS created_at
FROM work_team_member_links wtmf
WHERE wtmf.work_team_member_id = COALESCE(
  {{ appsmith.store.wtmRow?.work_team_member_id || 0 }},
  (
    SELECT w2.work_team_member_id
    FROM work_team_members w2
    WHERE COALESCE(w2.is_active,1) = 1
      AND w2.work_team_id = {{ appsmith.store.wtRow?.work_team_id || 0 }}
    ORDER BY w2.work_team_member_id ASC
    LIMIT 1
  )
)
AND COALESCE(wtmf.is_active,1) = 1
ORDER BY wtmf.wtm_link_id DESC;
