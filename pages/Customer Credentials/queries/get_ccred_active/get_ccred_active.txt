SELECT
  cr.customer_credential_id,
  cr.customer_id,
  c.customer_name,
  cr.customer_credential_name,
  cr.customer_credential_website,
  cr.customer_credential_notes,
  cr.customer_credential_status,
  cr.created_at,
  cr.updated_at
FROM customer_credentials cr
JOIN customers c ON c.customer_id = cr.customer_id
WHERE COALESCE(cr.customer_credential_status,'Active') = 'Active'
ORDER BY cr.customer_credential_id DESC;
