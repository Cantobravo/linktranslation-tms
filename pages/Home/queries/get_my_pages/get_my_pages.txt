SELECT DISTINCT
  ami.page_value
FROM resources r
JOIN roles ro
  ON ro.role_id = r.role_id
JOIN app_menu_items ami
  ON (ami.role_key COLLATE utf8mb4_unicode_ci)
     = (ro.role_name COLLATE utf8mb4_unicode_ci)
WHERE (r.resource_email COLLATE utf8mb4_unicode_ci)
      = (COALESCE(NULLIF({{ appsmith.user.email }}, ''), NULL)
         COLLATE utf8mb4_unicode_ci)
  AND ami.is_active = 1;
