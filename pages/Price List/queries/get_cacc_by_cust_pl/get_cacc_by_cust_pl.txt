SELECT
  customer_account_id   AS value,
  customer_account_name AS label
FROM customer_accounts
WHERE customer_id = COALESCE(
  {{
    (j_pl_upd_form.isVisible
      ? j_pl_upd_form.formData.customer_id
      : j_pl_ins_form.formData.customer_id) || null
  }},
  customer_id
)
AND COALESCE(customer_account_is_active,1) = 1
ORDER BY customer_account_name;
