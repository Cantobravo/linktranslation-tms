  price_list_name     = {{ j_pl_upd_form.formData.price_list_name }},
  currency_id         = {{ j_pl_upd_form.formData.currency_id }},
  customer_id         = CASE WHEN {{ j_pl_upd_form.formData.price_list_type }} = 'CUSTOMER'
                             THEN {{ j_pl_upd_form.formData.customer_id || null }}
                             ELSE NULL
                        END,
  pl_customer_account = CASE WHEN {{ j_pl_upd_form.formData.price_list_type }} = 'CUSTOMER'
                             THEN {{ j_pl_upd_form.formData.pl_customer_account || null }}
                             ELSE NULL
                        END,
  resource_id         = CASE WHEN {{ j_pl_upd_form.formData.price_list_type }} = 'RESOURCE'
                             THEN {{ j_pl_upd_form.formData.resource_id || null }}
                             ELSE NULL
                        END,
