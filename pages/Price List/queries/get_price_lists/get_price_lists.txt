SELECT
  pl.price_list_id,
  pl.price_list_type,
  pl.price_list_name,
  pl.currency_id,
  cur.currency_name,
  pl.customer_id,
  c.customer_name,
  pl.pl_customer_account,
  ca.customer_account_name,
  pl.resource_id,
  r.resource_full_name,
  pl.is_active,
  pl.created_at,
  pl.updated_at
FROM price_lists pl
LEFT JOIN currencies cur ON cur.currency_id = pl.currency_id
LEFT JOIN customers  c   ON c.customer_id     = pl.customer_id
LEFT JOIN customer_accounts ca ON ca.customer_account_id = pl.pl_customer_account
LEFT JOIN resources  r   ON r.resource_id     = pl.resource_id
ORDER BY pl.updated_at DESC;
