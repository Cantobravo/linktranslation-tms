SELECT
  pl.price_list_id,
  pl.price_list_type,
  pl.price_list_name,
  c.currency_name,
  pl.currency_id,
  cu.customer_name,
  pl.customer_id,
  r.resource_full_name,
  pl.resource_id,
  pl.is_active,
  pl.created_at,
  pl.updated_at
FROM price_lists pl
LEFT JOIN currencies c  ON c.currency_id  = pl.currency_id
LEFT JOIN customers cu  ON cu.customer_id = pl.customer_id
LEFT JOIN resources r   ON r.resource_id  = pl.resource_id
WHERE ( {{ sw_pl_show_inactive.isSwitchedOn ? 1 : 0 }} = 1 OR pl.is_active = 1 )
ORDER BY pl.updated_at DESC, pl.price_list_id DESC;
