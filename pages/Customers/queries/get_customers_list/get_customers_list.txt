SELECT
  c.customer_id,
  c.customer_name,
  c.customer_type,
  c.customer_country,
  cur.currency_name AS currency_name,
  c.customer_status,
  c.updated_at
FROM customers c
LEFT JOIN currencies cur
  ON cur.currency_id = c.currency_id
WHERE
  (
    {{ toggle_show_inactive_customers.isSwitchedOn ? 1 : 0 }} = 0
    AND c.customer_status = '10. Active Client'
  )
  OR
  (
    {{ toggle_show_inactive_customers.isSwitchedOn ? 1 : 0 }} = 1
    AND c.customer_status <> '10. Active Client'
  )
ORDER BY c.customer_id DESC;
