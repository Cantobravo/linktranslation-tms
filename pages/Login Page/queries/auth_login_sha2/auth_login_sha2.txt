SELECT
  r.resource_id,
  r.resource_username,
  r.resource_full_name,
  r.resource_email,
  r.role_id,
  r.resource_is_active
FROM resources r
WHERE r.resource_username = {{ i_username.text }}
  AND r.resource_is_active = 1
  AND r.resource_password_hash = SHA2({{ i_password.text }}, 256)
LIMIT 1;
